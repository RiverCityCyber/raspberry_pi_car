sudo apt update && upgrade
sudo lsblk -o NAME,SIZE,TYPE,MOUNTPOINT
sudo mkfs.ext4 -L ssd_data /dev/sda1

sudo mount /dev/sda1 /mnt/ssd
df -h | grep ssd
sudo nano /etc/fstab
  LABEL=ssd_data   /mnt/ssd   ext4   defaults,noatime   0   2
sudo umount /mnt/ssd
sudo mount -a
sudo chown -R pi:pi /mnt/ssd
